DISPLAY ("Enter a number:")
n <- int(INPUT ())

reverse_bin <- []

REPEAT UNTIL n = 0 {
	IF n MOD 2 = 1 {
		APPEND (reverse_bin, 1)
		n <- (n - 1) / 2
	}
	ELSE {
		APPEND (reverse_bin, 0)
		n <- n / 2
	}
}

DISPLAY ("Binary:")

index <- LENGTH (reverse_bin) - 1
REPEAT UNTIL index < 0 {
	DISPLAY (reverse_bin[index], end='')
	index <- index - 1
}
